Tornado Chart

Features:

  • Tornado Chart

    This chart type is set using the following code. 

    
//sets the Chart Type as Tornado
this.chartControl1.Series[0].Type = ChartSeriesType.Tornado;

  • Chart Axis Labels

    The axis labels of a chart can be set by handling the ChartFormatAxisLabel event.

    • 
string[] labels = new string[] {"", "100 + ", "95 to 99", "90 to 94", "10 to 14", "5 to 9", "Under 5"};

private void chartControl1_ChartFormatAxisLabel(object sender, ChartFormatAxisLabelEventArgs e)
        {
            if (e.AxisOrientation == ChartOrientation.Vertical)
            {
                int index = (int)e.Value;

                if (index >= 0 && index < labels.Length)
                {
                    //Applying custom label text for X Axis
                    e.Label = labels[index];
                }
                else
                {
                    e.Label = "";
                }
            }
            else
            {
                e.Label = "";
            }
            e.Handled = true;
        }
