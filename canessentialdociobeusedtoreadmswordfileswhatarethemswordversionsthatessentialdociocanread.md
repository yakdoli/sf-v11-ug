[·      ]{style="FONT-FAMILY: Symbol"}Word 97

[·      ]{style="FONT-FAMILY: Symbol"}Word 2000

[·      ]{style="FONT-FAMILY: Symbol"}Word 2002

[·      ]{style="FONT-FAMILY: Symbol"}Word 2003
