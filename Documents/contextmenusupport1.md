[·      ]{style="FONT-FAMILY: Symbol"}Paste

[·      ]{style="FONT-FAMILY: Symbol"}Paragraph

[·      ]{style="FONT-FAMILY: Symbol"}Bullets

[·      ]{style="FONT-FAMILY: Symbol"}Numbering
